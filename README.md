# pay-token-api

token rest api

#### Card Table

```
CREATE TABLE CARD (
    CARD_REF_ID BIGINT NOT NULL AUTO_INCREMENT,
    PAYER_CI VARCHAR(128) NOT NULL,
    PAYER_NAME VARCHAR(512) NOT NULL,
    CARD_CODE VARCHAR(512) NOT NULL,
    CARD_NUMBER VARCHAR(512) NOT NULL,
    CARD_VALID_YEAR VARCHAR(512) NOT NULL,
    CARD_VALID_MONTH VARCHAR(512) NOT NULL,
    CARD_CVS VARCHAR(512) NOT NULL,
    CREATED_TIME TIMESTAMP NOT NULL,
    PRIMARY KEY (CARD_REF_ID)
);
```

#### Token Table

```
CREATE TABLE TOKEN (
    TOKEN_ID BIGINT NOT NULL AUTO_INCREMENT,
    CARD_REF_ID BIGINT NOT NULL,
    TOKEN_TEXT VARCHAR(512) NOT NULL,
    TOKEN_EXPIRE TIMESTAMP NOT NULL,
    TOKEN_STATUS_CODE VARCHAR(16) NOT NULL,
    CREATED_TIME TIMESTAMP NOT NULL,
    PRIMARY KEY (TOKEN_ID)
);
```

#### Token status history Table

```
CREATE TABLE TOKEN_STATUS (
    TOKEN_STATUS_ID BIGINT NOT NULL AUTO_INCREMENT,
    TOKEN_ID BIGINT NOT NULL,
    TOKEN_STATUS_CODE VARCHAR(16) NOT NULL,
    CREATED_TIME TIMESTAMP NOT NULL,
    PRIMARY KEY (TOKEN_STATUS_ID)
);
```
